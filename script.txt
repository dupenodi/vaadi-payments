// ── One-time sheet setup (creates 9 tabs; run once) ───────────
function setupSheets() {
  var ss = SpreadsheetApp.openById(SHEET_ID);

  function ensureSheet(name, headers, colWidths, bgColor) {
    var sheet = ss.getSheetByName(name);
    if (sheet) ss.deleteSheet(sheet);
    sheet = ss.insertSheet(name);
    sheet.appendRow(headers);
    sheet.getRange(1, 1, 1, headers.length).setBackground(bgColor || '#1a1a2e').setFontColor('#ffffff').setFontWeight('bold');
    sheet.setFrozenRows(1);
    if (colWidths) colWidths.forEach(function(w, i) { sheet.setColumnWidth(i + 1, w); });
    return sheet;
  }

  // 1. Guests
  ensureSheet(SHEET_GUESTS, ['Guest ID', 'Name', 'Phone', 'Email', 'Notes', 'Created At', 'Last Used At'], [80, 140, 120, 140, 180, 140, 140], '#1a2e2e');

// ── Custom menu (shows when this script is bound to the sheet: Extensions > Apps Script) ─
function onOpen() {
  SpreadsheetApp.getUi()
    .createMenu('Vaadi')
    .addItem('Refresh Dashboard', 'refreshDashboard')
    .addItem('Setup Sheets (recreate all tabs)', 'menuSetupSheets')
    .addSeparator()
    .addItem('About', 'menuAbout')
    .addToUi();
}

function menuSetupSheets() {
  var ui = SpreadsheetApp.getUi();
  var ok = ui.alert('Setup Sheets', 'This will recreate all 9 tabs and delete existing data in them. Continue?', ui.ButtonSet.YES_NO);
  if (ok === ui.Button.YES) {
    setupSheets();
    ui.alert('Done', 'Sheets recreated. Refresh the spreadsheet if you don\'t see the new tabs.', ui.ButtonSet.OK);
  }
}

function menuAbout() {
  SpreadsheetApp.getUi().alert(
    'Vaadi Cabin Manager',
    'Reservations, Experiences, Food, Payments, Expenses.\n\nRefresh Dashboard: updates the Dashboard tab with current numbers.\n\nSetup Sheets: recreates all tabs (use only on a fresh sheet or when resetting).',
    SpreadsheetApp.getUi().ButtonSet.OK
  );
}

// ── Web app entry: GET = JSONP (no CORS), POST = JSON body (for images / large payloads)
function doGet(e) {
  var callback = e.parameter.callback;
  var payload  = e.parameter.payload;
  var result;
  try {
    var body  = JSON.parse(payload);
    var reply = runAgent(body.history || []);
    result = JSON.stringify({ ok: true, reply: reply, model: CONFIG.model });
  } catch(err) {
    result = JSON.stringify({ ok: false, error: err.message });
  }
  return ContentService
    .createTextOutput(callback + '(' + result + ')')
    .setMimeType(ContentService.MimeType.JAVASCRIPT);
}

function doPost(e) {
  var result;
  try {
    var body = {};
    if (e.postData && e.postData.contents) {
      var type = (e.postData.type || '').toLowerCase();
      var raw = e.postData.contents;
      if (type.indexOf('application/x-www-form-urlencoded') !== -1) {
        var params = {};
        raw.split('&').forEach(function(pair) {
          var i = pair.indexOf('=');
          if (i !== -1) {
            var k = decodeURIComponent(pair.slice(0, i).replace(/\+/g, ' '));
            var v = decodeURIComponent((pair.slice(i + 1) || '').replace(/\+/g, ' '));
            params[k] = v;
          }
        });
        body = params.payload ? JSON.parse(params.payload) : {};
      } else {
        body = JSON.parse(raw);
      }
    }
    var reply = runAgent(body.history || []);
    result = { ok: true, reply: reply, model: CONFIG.model };
  } catch(err) {
    result = { ok: false, error: err.message };
  }
  return ContentService
    .createTextOutput(JSON.stringify(result))
    .setMimeType(ContentService.MimeType.JSON)
    .setHeader('Access-Control-Allow-Origin', '*');
}

// ── Claude agentic loop (supports text + image content for vision) ─
function runAgent(history) {
  var messages = history.map(function(h) {
    var role = h.role === 'model' ? 'assistant' : h.role;
    if (Array.isArray(h.content)) {
      return { role: role, content: h.content };
    }
    var text = typeof h.content === 'string' ? h.content
             : (h.parts && h.parts[0] ? h.parts[0].text || '' : '');
    return { role: role, content: text };
  });

  for (var turn = 0; turn < 10; turn++) {
    var res = UrlFetchApp.fetch('https://api.anthropic.com/v1/messages', {
      method: 'post', contentType: 'application/json',
      headers: { 'x-api-key': ANTHROPIC_API_KEY, 'anthropic-version': '2023-06-01' },
      payload: JSON.stringify({
        model: CONFIG.model, max_tokens: CONFIG.max_tokens,
        temperature: CONFIG.temperature, system: SYSTEM_PROMPT,
        tools: TOOLS, messages: messages
      }),
      muteHttpExceptions: true
    });

    var data = JSON.parse(res.getContentText());
    if (data.error) throw new Error(data.error.message || JSON.stringify(data.error));

    if (data.stop_reason === 'end_turn') {
      var tb = data.content.filter(function(b){ return b.type==='text'; })[0];
      return tb ? tb.text : '(no response)';
    }

    if (data.stop_reason === 'tool_use') {
      messages.push({ role: 'assistant', content: data.content });
      var results = data.content
        .filter(function(b){ return b.type==='tool_use'; })
        .map(function(b){
          return { type:'tool_result', tool_use_id:b.id,
                   content: JSON.stringify(executeTool(b.name, b.input)) };
        });
      messages.push({ role: 'user', content: results });
      continue;
    }
    throw new Error('Unexpected stop_reason: ' + data.stop_reason);
  }
  return 'Agent loop exceeded max turns — please try again.';
}

function sheetToObjects(sheet) {
  var rows = sheet.getDataRange().getValues();
  if (rows.length <= 1) return [];
  var h = rows[0];
  return rows.slice(1).map(function(r){
    var o = {}; h.forEach(function(k,i){ o[k]=r[i]; }); return o;
  });
}

function objectsToRows(sheet, objects) {
  var headers = sheet.getRange(1,1,1,sheet.getLastColumn()).getValues()[0];
  return objects.map(function(o){ return headers.map(function(h){ return o[h]!==undefined?o[h]:''; }); });
}

function nextId(sheet, col, prefix) {
  var data = sheetToObjects(sheet);
  if (!data.length) return prefix+'001';
  var nums = data.map(function(r){ return parseInt(r[col].toString().replace(/\D/g,''))||0; });
  return prefix + String(Math.max.apply(null,nums)+1).padStart(3,'0');
}

function parseDate(s) {
  if (!s) return '';
  var d = new Date(s);
  return isNaN(d) ? s : d.toLocaleDateString('en-IN');
}

function toDate(s) {
  if (!s) return null;
  var d = new Date(s);
  return isNaN(d) ? null : d;
}

function nightsBetween(a,b) {
  var d1=new Date(a), d2=new Date(b);
  return (isNaN(d1)||isNaN(d2)) ? 1 : Math.max(1,Math.round((d2-d1)/86400000));
}

function toolAddBooking(a) {
  var d1 = toDate(a.check_in);
  var d2 = toDate(a.check_out);
  if (d1 && d2 && d2 <= d1) return { error: 'Check-out date must be after check-in date. Please correct the dates.' };
  var guestId = getOrCreateGuest(a.guest_name, a.phone || '');
  var sheet   = getRSV();
  var id     = nextId(sheet, 'Reservation ID', 'R');
  var nights = nightsBetween(a.check_in, a.check_out);
  var bookingType = a.booking_type || 'Forest View Cabin';
  var bookingSource = normaliseBookingSource(a.booking_source || 'Direct');
  var source = bookingSource.toLowerCase();
  var pricePerNight = Number(a.price_per_night) || 0;
  var discount = Number(a.total_discount) || 0;
  // Final Amount = (price/night × nights) - discount (or use provided final_amount)
  var finalAmount = a.final_amount != null ? Number(a.final_amount) : Math.round(pricePerNight * nights - discount);
  finalAmount = Math.max(0, finalAmount);
  // Tax 5% and Airbnb fee 15.5% when source is Airbnb; otherwise 0
  var tax = 0, airbnbFee = 0, netReceived = 0;
  if (a.tax != null && a.airbnb_fee != null && a.net_received != null) {
    tax = Number(a.tax) || 0;
    airbnbFee = Number(a.airbnb_fee) || 0;
    netReceived = Number(a.net_received) || 0;
  } else if (source.indexOf('airbnb') !== -1) {
    tax = Math.round(finalAmount * 0.05);
    airbnbFee = Math.round(finalAmount * 0.155);
    netReceived = a.net_received != null ? Number(a.net_received) : (finalAmount - tax - airbnbFee);
  } else {
    netReceived = a.net_received != null ? Number(a.net_received) : finalAmount;
  }
  netReceived = Math.max(0, netReceived);
  sheet.appendRow([
    id, guestId, a.guest_name, parseDate(a.check_in), bookingType, nights, bookingSource,
    pricePerNight, discount, finalAmount, tax, airbnbFee, netReceived,
    a.payout_date || '', parseDate(a.booking_date || a.check_in), Number(a.tip) || 0, a.payee || '',
    'reserved', a.notes || '', now(), now()
  ]);
  auditLog('system', 'reservation_created', 'reservation', id, JSON.stringify({ guest: a.guest_name, nights: nights, final_amount: finalAmount, tax: tax, airbnb_fee: airbnbFee, net_received: netReceived }));
  return { success: true, booking_id: id, reservation_id: id, guest: a.guest_name, guest_id: guestId, booking_type: bookingType, nights: nights, final_amount: finalAmount, tax: tax, airbnb_fee: airbnbFee, net_received: netReceived };
}

function toolEditBooking(a) {
  var booking = findBooking(a.booking_id);
  if (!booking) return { error:'Booking "'+a.booking_id+'" not found.' };

  var updates = {};
  if (a.guest_name) updates['Guest Name'] = a.guest_name;
  if (a.status) updates['Status'] = a.status;
  if (a.notes) updates['Notes'] = a.notes;
  if (a.check_in) updates['Check-in'] = parseDate(a.check_in);
  if (a.booking_type) updates['Booking Type'] = a.booking_type;
  if (a.nights != null) updates['Nights'] = Math.max(1, Number(a.nights));
  if (a.booking_source) updates['Booking Source'] = normaliseBookingSource(a.booking_source);
  if (a.display_price_per_night != null) updates['Display Price/Night'] = a.display_price_per_night;
  if (a.total_discount != null) updates['Total Discount'] = a.total_discount;
  if (a.final_amount != null) updates['Final Amount'] = a.final_amount;
  if (a.tax != null) updates['Tax (5%)'] = a.tax;
  if (a.airbnb_fee != null) updates['Airbnb Fee'] = a.airbnb_fee;
  if (a.net_received != null) updates['Net Received'] = a.net_received;
  if (a.payout_date != null) updates['Payout Date'] = a.payout_date;
  if (a.booking_date != null) updates['Booking Date'] = parseDate(a.booking_date);
  if (a.tip != null) updates['Tip'] = a.tip;
  if (a.payee != null) updates['Payee'] = a.payee;

  // When discount, price per night, or nights change: recompute Final Amount = (price × nights) - discount
  var pricePerNight = a.display_price_per_night != null ? Number(a.display_price_per_night) : (Number(booking['Display Price/Night']) || 0);
  var nights = a.nights != null ? Math.max(1, Number(a.nights)) : Math.max(1, Number(booking['Nights']) || 1);
  var totalDiscount = a.total_discount != null ? Number(a.total_discount) : (Number(booking['Total Discount']) || 0);
  if (a.total_discount != null || a.display_price_per_night != null || a.nights != null) {
    var derivedFinal = Math.max(0, Math.round(pricePerNight * nights - totalDiscount));
    updates['Final Amount'] = derivedFinal;
  }

  // Recalc tax (5%) and Airbnb fee (15.5%) when final_amount or booking_source changes
  var finalAmt = updates['Final Amount'] != null ? Number(updates['Final Amount']) : (a.final_amount != null ? Number(a.final_amount) : (Number(booking['Final Amount']) || 0));
  var src = (a.booking_source || booking['Booking Source'] || '').toString().toLowerCase();
  if (src.indexOf('airbnb') !== -1 && (updates['Final Amount'] != null || a.final_amount != null || a.booking_source != null || a.total_discount != null || a.display_price_per_night != null || a.nights != null)) {
    if (a.tax == null) { var taxVal = Math.round(finalAmt * 0.05); updates['Tax (5%)'] = taxVal; }
    if (a.airbnb_fee == null) { var feeVal = Math.round(finalAmt * 0.155); updates['Airbnb Fee'] = feeVal; }
    if (a.net_received == null) {
      var tax = updates['Tax (5%)'] != null ? updates['Tax (5%)'] : (Number(booking['Tax (5%)']) || 0);
      var fee = updates['Airbnb Fee'] != null ? updates['Airbnb Fee'] : (Number(booking['Airbnb Fee']) || 0);
      updates['Net Received'] = Math.max(0, finalAmt - tax - fee);
    }
  } else if ((src.indexOf('direct') !== -1 || src.indexOf('whatsapp') !== -1 || src.indexOf('instagram') !== -1) && (a.booking_source != null || updates['Final Amount'] != null)) {
    if (a.tax == null) updates['Tax (5%)'] = 0;
    if (a.airbnb_fee == null) updates['Airbnb Fee'] = 0;
    if (a.net_received == null) updates['Net Received'] = finalAmt;
  }

  if (Object.keys(updates).length === 0) return { error: 'Nothing to update — provide at least one field.' };
  updateBookingRow(booking['Reservation ID'], updates);
  return { success: true, booking_id: booking['Reservation ID'], guest: booking['Guest Name'], updated: updates };
}

function toolAddPayment(a) {
  var booking = findBooking(a.booking_id);
  if (!booking) return { error:'Booking "'+a.booking_id+'" not found.' };
  var amount = Number(a.amount)||0;
  var method = normaliseMethod(a.method);
  var finalAmt = Number(booking['Final Amount']) || 0;
  var netRecv  = Number(booking['Net Received']) || 0;
  var due      = Math.max(0, finalAmt - netRecv);

  if (method !== 'refund' && amount > due + 1) {
    return { error:'Payment Rs.'+amount+' exceeds due Rs.'+due+'. Please check.' };
  }

  var txnId = nextPaymentId();
  getPS().appendRow([txnId, booking['Reservation ID'], today(), amount, method, a.reference||'', a.payee||'', a.notes||'', now()]);
  // Update reservation Net Received so due reflects the new payment
  var newNetReceived = method === 'refund' ? Math.max(0, netRecv - amount) : (netRecv + amount);
  updateBookingRow(booking['Reservation ID'], { 'Net Received': newNetReceived });
  auditLog('system', 'payment_added', 'payment', txnId, JSON.stringify({ reservation_id: booking['Reservation ID'], method: method, amount: amount }));
  var tot = recalcBooking(booking['Reservation ID']);
  return { success: true, txn_id: txnId, booking_id: booking['Reservation ID'], method: method,
           amount_recorded: amount, total_paid: tot ? tot.paid : 0, remaining_due: tot ? tot.due : 0, fully_paid: tot ? tot.due===0 : false };
}

function toolGetReport(a) {
  var n = new Date();
  var from = toDate(a.from_date) || new Date(n.getFullYear(), n.getMonth(), 1);
  var to = toDate(a.to_date) || n;
  to.setHours(23, 59, 59);

  var reservations = sheetToObjects(getRSV()).filter(function(b){ var d = toDate(b['Check-in']); return d && d>=from && d<=to; });
  var experiences = sheetToObjects(getES()).filter(function(t){ var d = toDate(t['Date']); return d && d>=from && d<=to; });
  var food = sheetToObjects(getFS()).filter(function(t){ var d = toDate(t['Date']); return d && d>=from && d<=to; });
  var expenses = sheetToObjects(getXS()).filter(function(t){ var d = toDate(t['Date']); return d && d>=from && d<=to; });

  var stayRevenue = reservations.reduce(function(s,b){ return s+(Number(b['Net Received'])||0); }, 0);
  var airbnbRevenue = reservations.filter(function(b){ return (b['Booking Source']||'').toLowerCase().indexOf('airbnb')!==-1; }).reduce(function(s,b){ return s+(Number(b['Net Received'])||0); }, 0);
  var directRevenue = reservations.filter(function(b){ return (b['Booking Source']||'').toLowerCase().indexOf('airbnb')===-1; }).reduce(function(s,b){ return s+(Number(b['Net Received'])||0); }, 0);
  var experienceProfit = experiences.reduce(function(s,t){ return s+(Number(t['Net Profit'])||0); }, 0);
  var foodProfit = food.reduce(function(s,t){ return s+(Number(t['Food Profit'])||0); }, 0);
  var totalExpenses = expenses.reduce(function(s,t){ return s+(Number(t['Amount'])||0); }, 0);

  var byCategory = {};
  experiences.forEach(function(t){ var c=t['Experience Type']||'other'; byCategory[c]=(byCategory[c]||0)+(Number(t['Net Profit'])||0); });
  var expenseByCat = {};
  expenses.forEach(function(t){ var c=t['Category']||'other'; expenseByCat[c]=(expenseByCat[c]||0)+(Number(t['Amount'])||0); });

  var owing = sheetToObjects(getRSV()).filter(function(b){ return (Number(b['Final Amount'])||0) - (Number(b['Net Received'])||0) > 0; });

  return {
    success: true,
    period: { from: from.toLocaleDateString('en-IN'), to: to.toLocaleDateString('en-IN') },
    reservations_in_period: reservations.length,
    stay_revenue_net: stayRevenue,
    airbnb_revenue: airbnbRevenue,
    direct_revenue: directRevenue,
    experience_profit: experienceProfit,
    food_profit: foodProfit,
    total_expenses: totalExpenses,
    net_profit: stayRevenue + experienceProfit + foodProfit - totalExpenses,
    by_experience_type: byCategory,
    by_expense_category: expenseByCat,
    total_outstanding: owing.reduce(function(s,b){ return s + Math.max(0, (Number(b['Final Amount'])||0) - (Number(b['Net Received'])||0)); }, 0),
    reservations_with_dues: owing.length
  };
}

function toolGetOccupancy(a) {
  var n = new Date();
  var from = toDate(a.from_date) || n;
  var to = toDate(a.to_date) || new Date(n.getTime()+30*86400000);

  var data = sheetToObjects(getRSV()).filter(function(r){ return (r['Status']||'').toString().toLowerCase()!=='cancelled'; });

  function getCheckOut(r) {
    var ci = toDate(r['Check-in']); var nights = parseInt(r['Nights'],10)||1;
    if (!ci) return null;
    var co = new Date(ci); co.setDate(co.getDate() + nights);
    return co;
  }
  var inPeriod = data.filter(function(r){
    var ci = toDate(r['Check-in']), co = getCheckOut(r);
    return ci && co && ci<=to && co>=from;
  });

  var today_d = new Date(); today_d.setHours(0,0,0,0);
  var current = inPeriod.filter(function(r){ var ci=toDate(r['Check-in']), co=getCheckOut(r); return ci && co && ci<=today_d && co>today_d; });
  var upcoming = inPeriod.filter(function(r){ var ci=toDate(r['Check-in']); return ci && ci>today_d; });
  var checkedOut = inPeriod.filter(function(r){ var co=getCheckOut(r); return co && co<=today_d; });

  return { success: true,
    period: { from: from.toLocaleDateString('en-IN'), to: to.toLocaleDateString('en-IN') },
    current_guests: current.map(function(r){ var due = Math.max(0, (Number(r['Final Amount'])||0)-(Number(r['Net Received'])||0)); return { id: r['Reservation ID'], guest: r['Guest Name'], check_in: r['Check-in'], nights: r['Nights'], due: due }; }),
    upcoming: upcoming.map(function(r){ return { id: r['Reservation ID'], guest: r['Guest Name'], check_in: r['Check-in'], booking_type: r['Booking Type'] }; }),
    recently_checked_out: checkedOut.map(function(r){ var due = Math.max(0, (Number(r['Final Amount'])||0)-(Number(r['Net Received'])||0)); return { id: r['Reservation ID'], guest: r['Guest Name'], nights: r['Nights'], due: due }; })
  };
}

function toolCheckInBooking(a) {
  var booking = findBooking(a.booking_id);
  if (!booking) return { error:'Booking "'+a.booking_id+'" not found.' };
  updateBookingRow(booking['Reservation ID'], { 'Status': 'active' });
  auditLog('system', 'check_in', 'reservation', booking['Reservation ID'], '');
  return { success: true, booking_id: booking['Reservation ID'], guest: booking['Guest Name'], status: 'active', message: 'Marked as checked in.' };
}

function toolGetGuest(a) {
  var guest = getGuestById(a.guest_id);
  if (!guest) return { error:'Guest "'+a.guest_id+'" not found.' };
  var bookings = sheetToObjects(getRSV()).filter(function(r){ return r['Guest ID'] && r['Guest ID'].toString().toUpperCase()===guest['Guest ID'].toString().toUpperCase(); });
  var due = function(r){ return Math.max(0, (Number(r['Final Amount'])||0)-(Number(r['Net Received'])||0)); };
  return { success: true,
    guest: { id: guest['Guest ID'], name: guest['Name'], phone: guest['Phone'], email: guest['Email'], notes: guest['Notes'], created_at: guest['Created At'], last_used: guest['Last Used At'] },
    bookings_count: bookings.length,
    bookings: bookings.slice(0, 20).map(function(r){ return { id: r['Reservation ID'], check_in: r['Check-in'], booking_type: r['Booking Type'], nights: r['Nights'], status: r['Status'], due: due(r) }; })
  };
}

function toolListGuests(a) {
  var list = sheetToObjects(getGS());
  if (a.query && a.query.toString().trim()) {
    var q = a.query.toString().toLowerCase();
    list = list.filter(function(r){
      return (r['Name'] && r['Name'].toString().toLowerCase().indexOf(q)!==-1) ||
             (r['Phone'] && r['Phone'].toString().indexOf(q)!==-1) ||
             (r['Email'] && r['Email'].toString().toLowerCase().indexOf(q)!==-1);
    });
  }
  return { success:true, count:list.length,
    guests: list.map(function(r){
      return { id:r['Guest ID'], name:r['Name'], phone:r['Phone'], email:r['Email'], last_used:r['Last Used At'] };
    })};
}

function toolAddFood(a) {
  var r = findBooking(a.booking_id);
  if (!r) return { error: 'Reservation "' + a.booking_id + '" not found.' };
  var txnId = nextFoodId();
  var date = a.date ? parseDate(a.date) : today();
  var stayNights = Number(a.stay_nights)||1;
  var people = Number(a.people)||1;
  var rate = Number(a.rate_per_person_per_day)||0;
  var baseRev = Number(a.base_food_revenue)||(rate * people * stayNights);
  var cost = Number(a.food_cost)||0;
  var special = Number(a.special_request_charge)||0;
  var totalRev = baseRev + special;
  var profit = totalRev - cost;
  getFS().appendRow([
    txnId, r['Reservation ID'], date, a.guest || r['Guest Name'], stayNights, people, rate,
    baseRev, cost, special, totalRev, profit, a.payee || '', a.notes || '', now()
  ]);
  auditLog('system', 'food_added', 'food', txnId, JSON.stringify({ reservation_id: r['Reservation ID'], guest: a.guest, profit: profit }));
  return { success: true, txn_id: txnId, booking_id: r['Reservation ID'], guest: a.guest, total_food_revenue: totalRev, food_profit: profit };
}

function toolAddExpense(a) {
  var txnId = nextExpenseId();
  var date = a.date ? parseDate(a.date) : today();

  return { success: true, txn_id: txnId, category: category, amount: a.amount, paid_from: a.paid_from };
}

function toolGetDashboard(a) {
  var n = new Date();
  var from = toDate(a.from_date) || new Date(n.getFullYear(), n.getMonth(), 1);
  var to = toDate(a.to_date) || n;
  to.setHours(23, 59, 59);

  var allRes = sheetToObjects(getRSV());
  var allExp = sheetToObjects(getES());
  var allFood = sheetToObjects(getFS());
  var allExpenses = sheetToObjects(getXS());

  var reservations = allRes.filter(function(b){ var d = toDate(b['Check-in']); return d && d>=from && d<=to; });
  var experiences = allExp.filter(function(t){ var d = toDate(t['Date']); return d && d>=from && d<=to; });
  var food = allFood.filter(function(t){ var d = toDate(t['Date']); return d && d>=from && d<=to; });
  var expenses = allExpenses.filter(function(t){ var d = toDate(t['Date']); return d && d>=from && d<=to; });

  function sumNetReceived(arr) { return arr.reduce(function(s,b){ return s+(Number(b['Net Received'])||0); }, 0); }
  function sumProfit(arr) { return arr.reduce(function(s,t){ return s+(Number(t['Net Profit'])||0); }, 0); }
  function sumFoodProfit(arr) { return arr.reduce(function(s,t){ return s+(Number(t['Food Profit'])||0); }, 0); }
  function sumAmount(arr) { return arr.reduce(function(s,t){ return s+(Number(t['Amount'])||0); }, 0); }

  var stayRevenue = sumNetReceived(reservations);
  var airbnb = reservations.filter(function(b){ return (b['Booking Source']||'').toLowerCase().indexOf('airbnb')!==-1; });
  var direct = reservations.filter(function(b){ return (b['Booking Source']||'').toLowerCase().indexOf('airbnb')===-1; });
  var airbnbRev = sumNetReceived(airbnb);
  var directRev = sumNetReceived(direct);
  var expProfit = sumProfit(experiences);
  var foodProfit = sumFoodProfit(food);
  var totalExp = sumAmount(expenses);
  var netProfit = stayRevenue + expProfit + foodProfit - totalExp;

  var allStay = sumNetReceived(allRes);
  var allExpProfit = sumProfit(allExp);
  var allFoodProfit = sumFoodProfit(allFood);
  var allExpTotal = sumAmount(allExpenses);
  var allNet = allStay + allExpProfit + allFoodProfit - allExpTotal;

  var owing = allRes.filter(function(b){ return (Number(b['Final Amount'])||0) - (Number(b['Net Received'])||0) > 0; });
  var totalDue = owing.reduce(function(s,b){ return s + Math.max(0, (Number(b['Final Amount'])||0) - (Number(b['Net Received'])||0)); }, 0);

  function byBookingType(arr) {
    var sky = arr.filter(function(b){ return (b['Booking Type']||'').indexOf('Sky View')!==-1; });
    var forest = arr.filter(function(b){ return (b['Booking Type']||'').indexOf('Forest View')!==-1; });
    var villa = arr.filter(function(b){ return (b['Booking Type']||'').indexOf('Entire Villa')!==-1 || (b['Booking Type']||'').indexOf('Vaadi')!==-1; });
    return {
      sky_view: { count: sky.length, revenue: sumNetReceived(sky) },
      forest_view: { count: forest.length, revenue: sumNetReceived(forest) },
      entire_villa: { count: villa.length, revenue: sumNetReceived(villa) }
    };
  }

  function byExperienceType(arr) {
    var out = {};
    arr.forEach(function(t){
      var type = (t['Experience Type']||'other').toString();
      if (!out[type]) out[type] = { count: 0, total: 0, cost: 0, profit: 0 };
      out[type].count += 1;
      out[type].total += Number(t['Total Amount'])||0;
      out[type].cost += Number(t['Cost'])||0;
      out[type].profit += Number(t['Net Profit'])||0;
    });
    return out;
  }

  function byExpenseCategory(arr) {
    var out = {};
    arr.forEach(function(t){
      var cat = (t['Category']||'Misc').toString().trim() || 'Misc';
      out[cat] = (out[cat]||0) + (Number(t['Amount'])||0);
    });
    return out;
  }

  return { success: true,
    period: { from: from.toLocaleDateString('en-IN'), to: to.toLocaleDateString('en-IN') },
    summary_all_time: {
      total_reservations: allRes.length,
      stay_revenue_net: allStay,
      airbnb_revenue: sumNetReceived(allRes.filter(function(b){ return (b['Booking Source']||'').toLowerCase().indexOf('airbnb')!==-1; })),
      direct_revenue: sumNetReceived(allRes.filter(function(b){ return (b['Booking Source']||'').toLowerCase().indexOf('airbnb')===-1; })),
      experience_profit: allExpProfit,
      food_profit: allFoodProfit,
      total_expenses: allExpTotal,
      net_profit: allNet,
      outstanding_dues_count: owing.length,
      outstanding_dues_amount: totalDue
    },
    this_period: {
      reservations: reservations.length,
      stay_revenue_net: stayRevenue,
      airbnb_revenue: airbnbRev,
      direct_revenue: directRev,
      experience_profit: expProfit,
      food_profit: foodProfit,
      total_expenses: totalExp,
      net_profit: netProfit
    },
    by_booking_type: byBookingType(allRes),
    experiences_by_type: byExperienceType(allExp),
    expenses_by_category: byExpenseCategory(allExpenses),
    outstanding: { count: owing.length, total_amount: totalDue, bookings: owing.slice(0, 10).map(function(b){ return { id: b['Reservation ID'], guest: b['Guest Name'], due: Math.max(0, (Number(b['Final Amount'])||0)-(Number(b['Net Received'])||0)) }; }) }
  };
}

// Call from script editor or on a trigger to refresh Dashboard sheet with current numbers
function refreshDashboard() {
  var out = toolGetDashboard({});
  var sheet = getDS();
  var data = sheetToObjects(sheet);
  var headers = sheet.getRange(1, 1, 1, sheet.getLastColumn()).getValues()[0];
  var nowStr = new Date().toLocaleString('en-IN');
  var summary = out.summary_all_time;
  var byType = out.by_booking_type;
  var expByType = out.experiences_by_type;
  var expByCat = out.expenses_by_category;

  function setMetric(section, metric, value) {
    for (var i = 0; i < data.length; i++) {
      if (data[i]['Section'] === section && (data[i]['Metric']||'').toString().indexOf(metric) !== -1) {
        sheet.getRange(i + 2, headers.indexOf('Value') + 1).setValue(value);
        sheet.getRange(i + 2, headers.indexOf('Updated At') + 1).setValue(nowStr);
        return;
      }
    }
  }

  setMetric('Summary', 'Total Reservations', summary.total_reservations);
  setMetric('Summary', 'Stay Revenue', summary.stay_revenue_net);
  setMetric('Summary', 'Airbnb Revenue', summary.airbnb_revenue);
  setMetric('Summary', 'Direct Revenue', summary.direct_revenue);
  setMetric('Summary', 'Experience Profit', summary.experience_profit);
  setMetric('Summary', 'Food Profit', summary.food_profit);
  setMetric('Summary', 'Total Expenses', summary.total_expenses);
  setMetric('Summary', 'Net Profit', summary.net_profit);
  setMetric('Summary', 'Outstanding Dues (count)', summary.outstanding_dues_count);
  setMetric('Summary', 'Outstanding Dues (amount)', summary.outstanding_dues_amount);

  var mon = out.this_period;
  setMetric('Month', 'Reservations', mon.reservations);
  setMetric('Month', 'Stay Revenue', mon.stay_revenue_net);
  setMetric('Month', 'Experience Profit', mon.experience_profit);
  setMetric('Month', 'Food Profit', mon.food_profit);
  setMetric('Month', 'Expenses', mon.total_expenses);
  setMetric('Month', 'Net Profit', mon.net_profit);

  setMetric('By Type', 'Sky View Cabin (count)', byType.sky_view.count);
  setMetric('By Type', 'Sky View Cabin (revenue)', byType.sky_view.revenue);
  setMetric('By Type', 'Forest View Cabin (count)', byType.forest_view.count);
  setMetric('By Type', 'Forest View Cabin (revenue)', byType.forest_view.revenue);
  setMetric('By Type', 'Vaadi - Entire Villa (count)', byType.entire_villa.count);
  setMetric('By Type', 'Vaadi - Entire Villa (revenue)', byType.entire_villa.revenue);

  for (var type in expByType) {
    setMetric('Experiences', type, expByType[type].profit || expByType[type].total);
  }
  for (var cat in expByCat) {
    setMetric('Expenses', cat, expByCat[cat]);
  }

  Logger.log('Dashboard refreshed at ' + nowStr);
}
